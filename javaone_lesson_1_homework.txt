/*
* Написать метод, вычисляющий выражение a * (b + (c / d)) и возвращающий результат 
* с плавающей точкой, где a, b, c, d – целочисленные входные параметры этого метода;
*/

public static float calculateFormula(int a, int b, int c, int d) {     // d != 0
	float c1 = c;
	return a * (b + (c1 / d));
}



/*
* Написать метод, принимающий на вход два целых числа и проверяющий, что их сумма лежит
* в пределах от 10 до 20 (включительно), если да – вернуть true, в противном случае – false;
*/

public static boolean checkSum(int a, int b) {
	if ((a + b) >= 10 && (a + b) <= 20) {
		return true;
	} else {
		return false;
	}
}



/*
* Написать метод, которому в качестве параметра передается целое число, метод должен
* проверить, положительное ли число передали или отрицательное.
* Замечание: ноль считаем положительным числом. Результат работы метода вывести в консоль.
*/

public static boolean checkSign(int a) {	// метод, проверяющий целое число на положительность
	if (a >= 0) {
		return true;
	} else {
		return false;
	}
}

public static void printCheckSign(int a) {		// метод, выводящий в консоль результаты проверки
	if (checkSign(a)) {
		System.out.println("Число положительно");
	} else {
		System.out.println("Число отрицательно");
	}
}



/*
* Написать метод, которому в качестве параметра передается строка, обозначающая имя,
* метод должен вернуть приветственное сообщение «Привет, указанное_имя!». Вывести
* приветствие в консоль.
*/

public static String helloName(String name) {   // метод, генерирующий приветствие в виде строки
	String a = "Привет, ";
	String b = name + "!";
	return a + b;
}

public static void printHelloName(String name) {		// метод, выводящий приветствие в консоль
	System.out.println(helloName(name));
}



/*
* Написать метод, который определяет, является ли год високосным. Каждый 4-й год
* является високосным, кроме каждого 100-го, при этом каждый 400-й – високосный.
* Для проверки работы вывести результаты работы метода в консоль.
*/

public static boolean leapYear(int year) {						// проверка на високосность
	if ((year%4 != 0) || (year%100 == 0 && year%400 != 0)) {
		return false;
	} else {
		return true;
	}
}

public static void printLeapYear(int year) {				// метод, выводящий в консоль результаты проверки
	boolean bool = leapYear(year);				// переменная bool введена для удобства записи
	if (bool) {
		System.out.println(year + " год является високосным");
	} else {
		System.out.println(year + " год не является високосным");
	}
}
